280 string QueryPreprocessor::getToken(const string& query, int& pos)
281 {
282  int first = query.find_first_not_of(' ', pos);
283  if (first == string::npos)
284  return "";
285  pos = first + 1;
286  if ((query.at(first) >= 'A' && query.at(first) <= 'Z') ||
287  (query.at(first) >= 'a' && query.at(first) <= 'z')) { //IDENT
288  while (pos < query.length() && ((query.at(pos) >= 'A' && query.at(pos) <= 'Z') ||
289  (query.at(pos) >= 'a' && query.at(pos) <= 'z') || //letter or
290  (query.at(pos) >= '0' && query.at(pos) <= '9') || query.at(pos) == '#' || //digit
291  (query.at(pos) == '*') || //relation star
292  (query.at(pos) == '_' && pos == first + 4 && query.substr(first, 4) == "prog")))
293  pos++;
294  } else if (query.at(first) >= '0' && query.at(first) <= '9') { //DIGIT
295  while (pos < query.length() && (query.at(pos) >= '0' && query.at(pos) <= '9'))
296  pos++;
297  } else if (query.at(first) == '\"')
298  pos = query.find_first_of('\"', pos) + 1;
299  //else return single character
300  return query.substr(first, pos - first);
301 }
