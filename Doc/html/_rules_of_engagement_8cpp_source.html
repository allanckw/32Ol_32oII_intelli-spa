743  RulesOfEngagement::putNiceNiceASTNode(const RulesOfEngagement::QueryVar var,
744  const vector<ASTNode*>& answers)
745 {
746  vector<pair<int, unordered_set<ASTNode*>>> result;
747 
748  switch (var) {
749  case Procedure:
750  case Statement:
751  case Prog_Line: {
752  unordered_map<int, unordered_set<ASTNode*>> simplify;
753  for (auto it = answers.begin(); it != answers.end(); it++)
754  simplify[convertASTNodeToInteger(var, *it)].insert(*it);
755 
756  return vector<pair<int, unordered_set<ASTNode*>>>(simplify.begin(), simplify.end());
757  }
758  break;
759  case Assign:
760  for (auto it = answers.begin(); it != answers.end(); it++)
761  if (PKB::assignNodesBack.count(*it) > 0) {
762  unordered_set<ASTNode*> uset;
763  uset.insert(*it);
764  result.push_back(pair<int, unordered_set<ASTNode*>>
765  (PKB::assignNodesBack[*it], uset));
766  }
767  break;
768  case Call:
769  for (auto it = answers.begin(); it != answers.end(); it++)
770  if (PKB::callNodesBack.count(*it) > 0) {
771  unordered_set<ASTNode*> uset;
772  uset.insert(*it);
773  result.push_back(pair<int, unordered_set<ASTNode*>>
774  (PKB::callNodesBack[*it], uset));
775  }
776  break;
777  case While:
778  for (auto it = answers.begin(); it != answers.end(); it++)
779  if (PKB::whileNodesBack.count(*it) > 0) {
780  unordered_set<ASTNode*> uset;
781  uset.insert(*it);
782  result.push_back(pair<int, unordered_set<ASTNode*>>
783  (PKB::whileNodesBack[*it], uset));
784  }
785  break;
786  case If:
787  for (auto it = answers.begin(); it != answers.end(); it++)
788  if (PKB::ifNodesBack.count(*it) > 0) {
789  unordered_set<ASTNode*> uset;
790  uset.insert(*it);
791  result.push_back(pair<int, unordered_set<ASTNode*>>
792  (PKB::ifNodesBack[*it], uset));
793  }
794  break;
795  case Statement_List:
796  for (auto it = answers.begin(); it != answers.end(); it++)
797  if (PKB::statementListNodesBack.count(*it) > 0) {
798  unordered_set<ASTNode*> uset;
799  uset.insert(*it);
800  result.push_back(pair<int, unordered_set<ASTNode*>>
801  (PKB::statementListNodesBack[*it], uset));
802  }
803  break;
804  case Variable: {
805  unordered_map<int, unordered_set<ASTNode*>> simplify;
806  for (auto it = answers.begin(); it != answers.end(); it++)
807  if ((*it)->getType() == ASTNode::Variable)
808  simplify[convertASTNodeToInteger(var, *it)].insert(*it);
809  return vector<pair<int, unordered_set<ASTNode*>>>(simplify.begin(), simplify.end());
810  }
811  break;
812  case Constant: {
813  unordered_map<int, unordered_set<ASTNode*>> simplify;
814  for (auto it = answers.begin(); it != answers.end(); it++)
815  if ((*it)->getType() == ASTNode::Constant)
816  simplify[convertASTNodeToInteger(var, *it)].insert(*it);
817  return vector<pair<int, unordered_set<ASTNode*>>>(simplify.begin(), simplify.end());
818  }
819  break;
820  case Plus:
821  case Minus:
822  case Times:
823  for (auto it = answers.begin(); it != answers.end(); it++) {
824  unordered_set<ASTNode*> uset;
825  uset.insert(*it);
826  result.push_back(pair<int, unordered_set<ASTNode*>>(-1, uset));
827  }
828  break;
829  default:
830  throw new SPAException("Unknown type");
831  }
832 
833  return result;
834 }
