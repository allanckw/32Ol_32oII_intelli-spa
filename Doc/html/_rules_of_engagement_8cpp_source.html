1007 //RHS for now handles patterns in the form of "a" or _"a"_
1017 bool RulesOfEngagement::TryMatch(ASTNode* testedNode,
1018  RulesOfEngagement::PatternRHSType RHS, const ASTExprNode* pattern)
1019 {
1020  ASTNode* head = testedNode->getChild(1);
1021  stack<ASTNode*> nodesStack;
1022  nodesStack.push(head);
1023 
1024  if(RHS == RulesOfEngagement::PRSub) {
1025  while(nodesStack.size() > 0) {
1026  const ASTNode* tempnode = nodesStack.top();
1027  nodesStack.pop();
1028 
1029  if (MatcherTree(tempnode,pattern))
1030  return true;
1031 
1032  if(tempnode->getType() == ASTNode::Operator) {
1033  nodesStack.push(tempnode->getChild(1));//add right side in
1034  nodesStack.push(tempnode->getChild(0));//add left side in
1035  }
1036  }
1037  return false;
1038  } else
1039  return MatcherTree(head,pattern);
1040 }

1050 {

1052  Original->getValue() == Pattern->getValue() &&
1053  (Original->getType() == ASTNode::Constant ||

1055  (Original->getType() == ASTNode::Operator &&
1056  MatcherTree(Original->getChild(0), Pattern->getChild(0)) &&
