502 //RHS for now handles patterns in the form of "a" or _"a"_
503 bool RulesOfEngagement::TryMatch(ASTNode* testedNode,
504  RulesOfEngagement::PatternRHSType RHS, const ASTExprNode* pattern)
505 {
506  ASTNode* head = testedNode->getChild(1);
507  stack<ASTNode*> nodesStack;
508  nodesStack.push(head);
509 
510  if(RHS == RulesOfEngagement::PRSub) {
511  while(nodesStack.size() > 0) {
512  const ASTNode* tempnode = nodesStack.top();
513  nodesStack.pop();
514 
515  if (MatcherTree(tempnode,pattern))
516  return true;
517 
518  if(tempnode->getType() == ASTNode::Operator) {
519  nodesStack.push(tempnode->getChild(1));//add right side in
520  nodesStack.push(tempnode->getChild(0));//add left side in
521  }
522  }
523  return false;
524  } else
525  return MatcherTree(head,pattern);
526 }

528 bool RulesOfEngagement::MatcherTree(const ASTNode* Original, const ASTNode* Pattern)//, bool isSub)
529 {
