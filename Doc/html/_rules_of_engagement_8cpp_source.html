800 //RHS for now handles patterns in the form of "a" or _"a"_
810 bool RulesOfEngagement::TryMatch(ASTNode* testedNode,
811  RulesOfEngagement::PatternRHSType RHS, const ASTExprNode* pattern)
812 {
813  ASTNode* head = testedNode->getChild(1);
814  stack<ASTNode*> nodesStack;
815  nodesStack.push(head);
816 
817  if(RHS == RulesOfEngagement::PRSub) {
818  while(nodesStack.size() > 0) {
819  const ASTNode* tempnode = nodesStack.top();
820  nodesStack.pop();
821 
822  if (MatcherTree(tempnode,pattern))
823  return true;
824 
825  if(tempnode->getType() == ASTNode::Operator) {
826  nodesStack.push(tempnode->getChild(1));//add right side in
827  nodesStack.push(tempnode->getChild(0));//add left side in
828  }
829  }
830  return false;
831  } else
832  return MatcherTree(head,pattern);
833 }

842 bool RulesOfEngagement::MatcherTree(const ASTNode* Original, const ASTNode* Pattern)//, bool isSub)
843 {
844  return (Original->getType() == Pattern->getType() &&
845  Original->getValue() == Pattern->getValue() &&
846  (Original->getType() == ASTNode::Constant ||
847  Original->getType() == ASTNode::Variable ||
848  (Original->getType() == ASTNode::Operator &&
849  MatcherTree(Original->getChild(0), Pattern->getChild(0)) &&
850  MatcherTree(Original->getChild(1), Pattern->getChild(1)))));
851 }
